2회차 숙제
```js
const generator = function*(i, j){
  for (let n = 1; n <= i; n++) {
    for (let nn = 1; nn <= j; nn++) {
      yield [n, nn, n * nn]
    }
  }
};

for(const [i, j, k] of generator(9,9)){
  console.log(`${i} x ${j} = ${k}`);
}

```
